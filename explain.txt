1,create  webpack.config.js  file

        const path = require('path');

        //entry 是相对路径。从entry得到代码。从output输出代码。
        //path 是要绝对路径
        // module.exports必须在node.js里面才能执行
        module.exports = {
            entry: './src/index.js',
            output: {
                path: path.resolve(__dirname, 'dist'),
                filename: 'bundle.js'
            }
        };


2， npm install webpack webpack-cli --save-dev

3, run webpack commend line 

   node_modules/.bin/webpack

   这样src/index.js的文件会被翻译到 dist/assets/bundle.js文件里去。
   注意这时webpack并不是像babel那样，将现代javascript代码翻译成所有浏览器都能解析的代码。
   每次修改文件后都要执行 node_modules/.bin/webpack 会很麻烦
    在package.json里面 写上"webpack": "node_modules/.bin/webpack"

    "scripts": {
        "babel": "node_modules/.bin/babel ./src/index.js -w -o ./dist/assets/bundle.js",
        "webpack": "node_modules/.bin/webpack"
    },


   然后 npm run webpack 即可简单快捷地翻译代码（注意并不是实时的翻译）


4,dom.js里的代码被import到 index.js文件，然后index.js的文件被翻译到 bundle.js文件里去。
  
5，然后打开 http://127.0.0.1:5500/dist/ 就可以看到。index.html 文件的效果了。
